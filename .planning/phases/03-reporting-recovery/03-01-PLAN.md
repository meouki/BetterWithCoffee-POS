### 3-03: Replace `sequelize.sync()` with Umzug Runner [Depends on 3-02]
- **Action**:
    1. In `backend_rewrite/server.js`, replace the `sequelize.sync()` call with the `umzug` runner:
       ```js
       const { Umzug, SequelizeStorage } = require('umzug');
       const umzug = new Umzug({
           migrations: { glob: 'migrations/*.js', resolve: ({ name, path: mp, context }) => {
               const migration = require(mp);
               return { name, up: async () => migration.up(context, require('sequelize')), down: async () => migration.down(context, require('sequelize')) };
           }},
           context: sequelize.getQueryInterface(),
           storage: new SequelizeStorage({ sequelize }),
           logger: console,
       });
       
       umzug.up()
           .then(async () => {
               console.log('✅ Migrations applied.');
               await seedDatabase();
               // ... rest of startup (tunnel + listen)
           })
           .catch(err => console.error('❌ Migration failed:', err));
       ```
    2. Keep `sequelize.sync({ force: true })` INTACT inside `routes/system.js` (it's the intentional wipe route).
- **Read First**: `backend_rewrite/server.js`, `backend_rewrite/routes/system.js`.
- **Acceptance Criteria**:
    - `sequelize.sync()` no longer exists in `server.js` (only `umzug.up()`).
    - `sequelize.sync({ force: true })` in `routes/system.js` is untouched.
    - Starting the server with `node server.js` logs "✅ Migrations applied." before the server port message.

           }
       }
